# ── 1. LOAD LIBRARIES ────────────────────────────────────────────────────────
# No external libraries needed for data parsing; base R handles this perfectly.

# ── 2. SET PATHS AND IMPORT FILE ─────────────────────────────────────────────
# Replace with your actual directory path and filename
file_path <- "C:/Users/Hana/Desktop/Screening/data_RNASeq/demo_count.txt" 

# Read the file (featureCounts outputs a tab-delimited file with a 1-line header comment)
fc_data <- read.table(
  file = file_path, 
  header = TRUE, 
  sep = "\t", 
  stringsAsFactors = FALSE, 
  comment.char = "#"  # Automatically skips the metadata comment line generated by featureCounts
)

# Extract gene IDs, gene lengths, and the raw count column (column 7)
gene_ids <- fc_data$Geneid
gene_lengths <- fc_data$Length
raw_counts <- fc_data[, 7]

# ── 2. METRIC 1: GLOBAL READ SUMS & GENE DETECTION RATES ────────────────────
total_assigned_reads <- sum(raw_counts)
total_genes_in_ref   <- length(raw_counts)

# Quantify detected genes using standard detection thresholds
detected_greater_0  <- sum(raw_counts > 0)
detected_greater_10 <- sum(raw_counts >= 10)

# Calculate percentage of reference library active in this sample
pct_detected_gt0 <- (detected_greater_0 / total_genes_in_ref) * 100

# Print Text Summary to Console (Fixed using native R syntax)
cat(sprintf("Total Assigned Reads   : %d\n", total_assigned_reads))
cat(sprintf("Total Genes in Ref     : %d\n", total_genes_in_ref))
cat(sprintf("Genes Detected (>0)    : %d (%.2f%%)\n", detected_greater_0, pct_detected_gt0))
cat(sprintf("Genes Highly Expr (>=10): %d\n", detected_greater_10))

# ── 3. METRIC 2: QUANTILE DISTRIBUTION SPREAD ────────────────────────────────
print(quantile(raw_counts, probs = c(0, 0.25, 0.5, 0.75, 0.9, 0.95, 0.99, 1)))

# ── 4. METRIC 3: DISTRIBUTION PLOTS ──────────────────────────────────────────
# Setup side-by-side plotting area
par(mfrow = c(1, 2))

# Plot A: Boxplot of log2 counts (adding pseudo-count of 1 to handle zeroes)
boxplot(log2(raw_counts + 1), 
        main = "Log2 Count Distribution", 
        ylab = "log2(Counts + 1)", 
        col = "skyblue")

# Plot B: Density distribution profile of expressed genes only
expressed_counts <- raw_counts[raw_counts > 0]
plot(density(log2(expressed_counts)), 
     main = "Expressed Genes Density Plot", 
     xlab = "log2(Counts)", 
     col = "darkblue", 
     lwd = 2)

# Reset plotting window defaults
par(mfrow = c(1, 1))

# Find the indices of the top 10 highest counts
top_10_indices <- order(raw_counts, decreasing = TRUE)[1:10]

# Extract their metadata and raw counts
top_10_genes <- data.frame(
  Geneid = gene_metadata$Geneid[top_10_indices],
  Length = gene_metadata$Length[top_10_indices],
  Raw_Count = raw_counts[top_10_indices],
  Pct_Total_Library = round((raw_counts[top_10_indices] / total_assigned_reads) * 100, 2)

print(top_10_genes)

write.table(top_10_genes, 
            file = "top10_genes.txt", 
            sep = '\t',
            row.names=FALSE, quote = TRUE)

# ── 1. MAP MITOCHONDRIAL GENES ───────────────────────────────────────────────
# Check how many genes originate from the mitochondrial chromosome
mt_genes_mask <- gene_metadata$Chr == "chM" | gene_metadata$Chr == "MT"

# If your annotation uses Gene Symbols instead of Ensembl IDs in the rows:
# mt_genes_mask <- grepl("^MT-", gene_metadata$Geneid, ignore.case = TRUE)

total_mt_counts <- sum(raw_counts[mt_genes_mask])
pct_mt <- (total_mt_counts / total_assigned_reads) * 100

# ── 2. PRINT MITOCHONDRIAL CONTINUATION REPORT ───────────────────────────────
cat(sprintf("Total Mitochondrial Reads: %d\n", total_mt_counts))
cat(sprintf("Mitochondrial Content %% : %.2f%%\n", pct_mt))

# ── 3. VISUALIZE READ LENGTH VS COUNT RELATIONSHIP ───────────────────────────
# Check if longer transcripts are collecting more reads artificially
expressed_idx <- raw_counts > 0
length_corr <- cor(gene_metadata$Length[expressed_idx], raw_counts[expressed_idx], method = "spearman")

plot(log10(gene_metadata$Length[expressed_idx]), log2(raw_counts[expressed_idx]),
     xlab = "log10(Gene Length bp)", ylab = "log2(Raw Counts)",
     main = sprintf("Length Bias (Spearman rho = %.3f)", length_corr), 
     col = rgb(0.1, 0.2, 0.8, 0.2), pch = 16)
abline(lm(log2(raw_counts[expressed_idx]) ~ log10(gene_metadata$Length[expressed_idx])), col = "darkred", lwd = 2)
